ポップオーバーの表示
ポップオーバーはpopoverメソッドを使って設定される。
placement：位置（top,bottom,left,rightのいずれか）
title：ポップオーバーのタイトルテキスト
content：ポップオーバーのコンテンツ
■scriptの内容 
$(function () {// DOM構造が解析された後に動く

    $('#ID')// 指定したIDの要素を取得
    .popover({'placement': 'bottom', 'title': 'This is Popover!',
            'content': 'This is popover sample content.'})// 引数にplacement,title,contentを指定し、popoverを生成

});
フォーカスによる非表示
本来のポップオーバーはボタンを再度クリックしないと
ポップオーバーが消えなくなっている。
data-trigger属性にfocusを指定することで
ボタン以外の場所をクリックするだけでポップオーバーを非表示にできる。
■scriptの内容 
$(function () {// DOM構造が解析された後に動く

    $('#ID')// 指定したIDの要素を取得
    .popover({'placement': 'bottom', 'trigger': 'focus', 'title': 'This is Popover!',
            'content': 'This is popover sample content.'})// 引数にplacement,title,content,triggerを指定し、popoverを生成

});
HTMLを利用したポップオーバーの表示
ポップオーバーの内容にHTMLを操作しなくてもscriptの操作のみで可能。
引数に指定するもので変数名が入る場合は「''」は不要である。
'html':trueを指定するとtitleやcontentにHTMLコードを記述出来るようになる。（「''」不要）
■scriptの内容 
$(function () {// DOM構造が解析された後に動く

    (function () {// 即時関数を利用
        let img = '<img id="popoverImg" src="../img/sample.jpg" alt="sample" style="width: 75px;">';// 変数imgにimgタグを代入
        let title = '<div>This is popover!</div>';// 変数titleに代入
        let content = 'This is <b>popover</b> sample content.';// 変数contentに代入
        let html = '<div class="row"><div class="col-4">' + img + '</div>' + '<div class="col-8 lead">' + content + '</div></div>';// 変数htmlに代入
        $('#popoverBtn3 > button').popover({// 引数を指定し、popoverを生成
            'placement': 'top',
            'trigger': 'focus',
            'title': title,
            'html': true,
            'content': $($.parseHTML(html)),
            // JQueryオブジェクトで渡さないと、imgのstyleが無視される（多分Bootstrapのバグ）
            'content':html, 本来はこれでいけるはず。
        });
    })();

});
