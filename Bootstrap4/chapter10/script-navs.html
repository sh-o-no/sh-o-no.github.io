Navのタブ表示を切り替える
Navのタブ表示を切り替えたい場合にはtebメソッドを使うことで可能となる。
tabメソッドの引数に'show'を指定して呼び出すとそのタブが選択される。
下記のサンプルはwindows以外を選択している場合でもボタンをクリックするとwindowsタブに戻る。
■Scriptの内容 
// 要素をクリックしたときの処理
$('<selector>').click(function () {
    $('<tab>').tab('show');// 指定のタブを選択する
});

タブの切替イベント
タブの切替に関する独自のイベントが4種類用意されている。
show.bs.tab：タブが選択された（切り替わりの開始）
shown.bs.tab：タブが選択された状態に切り替わった（切り替わりの完了）
hide.bs.tab：タブが非選択になった（切り替わりの開始）
hidden.bs.tab：タブが非選択状態に切り替わった（切り替わりの完了）
■Scriptの内容 
(function () {
    // タブの背景色をつける
    let tab_list = $('<selector>');
    tab_list.addClass('bg-info text-dark');// 要素にクラスを付け加える
    $('#tab2 a.active')
        .removeClass('bg-info text-dark')// クラスを削除
        .addClass('bg-primary text-white');// 更にクラスを付け加える

    // 指定のタブが選択されたときの処理
    tab_list.on('show.bs.tab',function (event) {
        $(event.target)// そのイベントが起こった要素に
            .addClass('bg-primary text-white');// クラスを付け加える
            .removeClass('bg-info text-dark');// 更にクラスの削除
    });

    // 指定のタブが非表示になったときの処理
    tab_list.on('hide.bs.tab',function (event) {
        $(event.target)// そのイベントが起こった要素に
            .addClass('bg-info text-dark');// クラスを付け加える
            .removeClass('bg-primary text-white');// 更にクラスの削除
    });
})();
