ジャンボトロンの表示
ジャンボトロンは大きな画面で表示させるようなインパクトのあるコンテンツなので
常時表示されていたら邪魔になる場合がある。
そういった場合にクラスを操作し、ジャンボトロンにしたり
そうでなくしたりすることが出来る。
■Scriptの構成 
let flg = false;
$('ID').click(function () {// 指定のIDの要素をクリックしたときの処理
    flg = !flg;
    if (flg) {// もしtrueなら各タグにクラスを追加
        $('#jumbo').addClass('jumbotron jumbotron-fluid');
        $('#jumbo div').addClass('container');
        $('#jumbo div h3').addClass('display-3');
        $('#jumbo div p').addClass('lead');
    } else {// そうでないなら各タグからクラスを削除
        $('#jumbo').removeClass('jumbotron jumbotron-fluid');
        $('#jumbo div').removeClass('container');
        $('#jumbo div h3').removeClass('display-3');
        $('#jumbo div p').removeClass('lead');
     }
});
